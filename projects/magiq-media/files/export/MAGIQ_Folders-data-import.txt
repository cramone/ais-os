CREATE TABLE dbo.MEDIA_FOLDERS_STAGE
(
    FOLDER_ID nvarchar(100),
    FOLDER_NAME nvarchar(1000),
    FOLDER_DESCRIPTION nvarchar(4000),
    FOLDER_REGISTERDATE nvarchar(100),
    FOLDER_OWNER nvarchar(1000),
    FOLDER_PATH nvarchar(4000)
);

BULK INSERT dbo.MEDIA_FOLDERS_STAGE
FROM '/var/opt/mssql/data/MAGIQ_Folders-data.csv'
WITH
(
    FORMAT = 'CSV',
    FIRSTROW = 2
);

SELECT *
FROM dbo.MEDIA_FOLDERS_STAGE
WHERE TRY_CONVERT(datetime, FOLDER_REGISTERDATE) IS NULL
  AND FOLDER_REGISTERDATE IS NOT NULL
  AND LTRIM(RTRIM(FOLDER_REGISTERDATE)) <> '';


  INSERT INTO dbo.MEDIA_FOLDERS
(
    FOLDER_ID,
    FOLDER_NAME,
    FOLDER_DESCRIPTION,
    FOLDER_REGISTERDATE,
    FOLDER_OWNER,
    FOLDER_PATH
)
SELECT
    FOLDER_ID,
    FOLDER_NAME,
    FOLDER_DESCRIPTION,
    TRY_CONVERT(datetime, NULLIF(FOLDER_REGISTERDATE, '')),
    FOLDER_OWNER,
    FOLDER_PATH
FROM dbo.MEDIA_FOLDERS_STAGE;
